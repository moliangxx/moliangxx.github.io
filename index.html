C-Mopao-Sort

#include
#include
using namespace std;

int main(){
ios::sync_with_stdio(false);
cin.tie(nullptr);
cout.tie(nullptr);

int n;
cin>>n;
vector num(n);
for(int i=0;i<n;i++){
cin>>num[i];
}
for(int i=1;i<n;i++){
bool flag = true;
for(int j=1;j<n-i;j++){
if(num[j-1]>num[j]){
int c=num[j-1];
num[j-1] = num[j];
num[j] = c;
flag =false;
}
}
if(flag){
break;
}
}
for(int i=0;i<n;i++){
cout<<num[i]<<” “;
}
return 0;
}
